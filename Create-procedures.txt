Листинг запроса для одновременного добавления новой партии удобрений и обновление остатка на складе. Это гарантирует, что данные в таблицах batches (партии) и stock (остатки) всегда будут согласованы. Выполнение этих двух действий в одной процедуре делает операцию атомарной — либо обе записи создаются успешно, либо, в случае ошибки, не создаётся ни одна:
CREATE OR REPLACE PROCEDURE add_batch_and_update_stock(
    p_fertilizer_id INTEGER,
    p_supplier_id INTEGER,
    p_warehouse_id INTEGER,
    p_batch_number VARCHAR(100),
    p_production_date DATE,
    p_expiration_date DATE,
    p_quantity NUMERIC(15,2),
    p_price_per_unit NUMERIC(12,2)
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_total_price NUMERIC(15,2);
BEGIN
    -- общая стоимость партии
    v_total_price := p_quantity * p_price_per_unit;
    -- вставка новой запии о партии
    INSERT INTO batches (
        fertilizer_id, supplier_id, warehouse_id, batch_number,
        production_date, expiration_date, quantity, price_per_unit, total_price
    ) VALUES (
        p_fertilizer_id, p_supplier_id, p_warehouse_id, p_batch_number,
        p_production_date, p_expiration_date, p_quantity, p_price_per_unit, v_total_price
    );

    -- обновить существующую запись
    UPDATE stock
    SET quantity = quantity + p_quantity
    WHERE warehouse_id = p_warehouse_id AND fertilizer_id = p_fertilizer_id;
    -- если запись для этого удобрения на этом складе ещё не существует 
    IF NOT FOUND THEN
        -- вставка новой записи в таблицу остатков
        INSERT INTO stock (warehouse_id, fertilizer_id, quantity)
        VALUES (p_warehouse_id, p_fertilizer_id, p_quantity);
    END IF;
END;
$$;

Листинг запроса по созданию процедуры которая автоматизирует процесс инвентаризации и списания товаров, срок годности которых истёк:
CREATE OR REPLACE PROCEDURE write_off_expired_batches()
LANGUAGE plpgsql
AS $$
DECLARE
    expired_batch RECORD;
BEGIN
    FOR expired_batch IN
        SELECT b.id, b.fertilizer_id, b.warehouse_id, b.quantity
        FROM batches b
        WHERE b.expiration_date < CURRENT_DATE
    LOOP
        INSERT INTO outcome_operations (warehouse_id, fertilizer_id, quantity, purpose)
        VALUES (
            expired_batch.warehouse_id,
            expired_batch.fertilizer_id,
            expired_batch.quantity,
            'Списание по истечении срока годности'
        );
        DELETE FROM batches WHERE id = expired_batch.id;
    END LOOP;
END;
$$;


Листинг запроса где процедура позволяет перемещать товар с одного склада на другой:
CREATE OR REPLACE PROCEDURE transfer_stock_between_warehouses(
    p_fertilizer_id INTEGER,
    p_from_warehouse_id INTEGER,
    p_to_warehouse_id INTEGER,
    p_quantity NUMERIC(15,2)
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_current_qty NUMERIC(15,2);
BEGIN
    SELECT quantity INTO v_current_qty
    FROM stock
    WHERE warehouse_id = p_from_warehouse_id AND fertilizer_id = p_fertilizer_id;
    IF v_current_qty IS NULL OR v_current_qty < p_quantity THEN
        RAISE EXCEPTION 'Недостаточно товара (ID=%) на складе (ID=%) для переноса. Доступно: %', p_fertilizer_id, p_from_warehouse_id, COALESCE(v_current_qty, 0);
    END IF;
    UPDATE stock
    SET quantity = quantity - p_quantity
    WHERE warehouse_id = p_from_warehouse_id AND fertilizer_id = p_fertilizer_id;
    UPDATE stock
    SET quantity = quantity + p_quantity
    WHERE warehouse_id = p_to_warehouse_id AND fertilizer_id = p_fertilizer_id;
    IF NOT FOUND THEN
        INSERT INTO stock (warehouse_id, fertilizer_id, quantity)
        VALUES (p_to_warehouse_id, p_fertilizer_id, p_quantity);
    END IF;
END;
$$;
